/* leap year checker */
int main()
{
  int year;

  printf("Enter a year to check if it is a leap year\n");

  if ( year%400 == 0)
    println("%d is a leap year.\n", year);
  else if ( year%100 == 0)
    println("%d is not a leap year.\n", year);
  else if ( year%4 == 0 )
    println("%d is a leap year.\n", year);
  else
    println("%d is not a leap year.\n", year);  
 
  return 0;
}
